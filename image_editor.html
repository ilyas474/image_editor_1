import java.io.File; public class CreateAndDeleteFolder { public static void main(String[] args) { File folder = new File("MyFolder"); if (folder.mkdir()) { System.out.println("Folder created."); } else { System.out.println("Folder already exists or failed to create."); } if (folder.delete()) { System.out.println("Folder deleted."); } else { System.out.println("Failed to delete the folder."); } } }
